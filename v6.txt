    function defaultState() {
      return {
        currentPage: 0,

    function normalizeImageData(page) {
      if (!('image' in page)) {
        if (typeof page.textScale !== 'number') page.textScale = 1;
        return page;
      }

      if (!page.image || typeof page.image === 'string') {
        page.image = {
          src: typeof page.image === 'string' ? page.image : '',
          zoom: 1,
          x: 0,
          y: 0
        };
      }

      if (typeof page.image.zoom !== 'number') page.image.zoom = 1;
      if (typeof page.image.x !== 'number') page.image.x = 0;
      if (typeof page.image.y !== 'number') page.image.y = 0;
      if (typeof page.image.src !== 'string') page.image.src = '';
      if (typeof page.textScale !== 'number') page.textScale = 1;
      return page;
    }

    function loadState() {
      const raw = localStorage.getItem(STORAGE_KEY);
      if (!raw) return defaultState();

      try {
        const parsed = JSON.parse(raw);
        if (!Array.isArray(parsed.pages) || parsed.pages.length === 0) return defaultState();
        parsed.pages = parsed.pages.map(page => normalizeImageData(page));
        if (typeof parsed.currentPage !== 'number') parsed.currentPage = 0;
        if (parsed.currentPage >= parsed.pages.length) parsed.currentPage = parsed.pages.length - 1;
        return parsed;
      } catch (error) {
        console.error(error);
        return defaultState();
      }
    }

    function setSaveStatus(text) {

    }

    function queueSave() {
      setSaveStatus('저장 중...');
      clearTimeout(saveTimer);
      saveTimer = setTimeout(() => {
        try {
          localStorage.setItem(STORAGE_KEY, JSON.stringify(appState));
          setSaveStatus('자동저장 완료');
        } catch (error) {
          console.error(error);
          setSaveStatus('저장 실패');
        }
      }, 180);
    }

    function escapeHtml(value = '') {
      return String(value)
        .replaceAll('&', '&amp;')
        .replaceAll('<', '&lt;')
        .replaceAll('>', '&gt;')
        .replaceAll('"', '&quot;')
        .replaceAll("'", '&#039;');
    }

    function multilineToHtml(value = '') {
      return escapeHtml(value).replace(/\n/g, '<br>');
    }

    function getLayoutLabel(value) {
      return layoutOptions.find(item => item.value === value)?.label || value;
    }

    function getPageDisplayTitle(page, index) {
      return page.koreanTitle || page.title || page.copy || `Page ${index + 1}`;
    }

    function getImage(page) {
      if (!('image' in page)) return null;
      return normalizeImageData(page).image;
    }

    function imageTransformStyle(image) {
      const zoom = image?.zoom ?? 1;
      const x = image?.x ?? 0;
      const y = image?.y ?? 0;
      return `transform: translate(calc(-50% + ${x}px), calc(-50% + ${y}px)) scale(${zoom});`;
    }

    function renderImageBox(image, extraClass = '') {
      if (!image || !image.src) {
        return `<div class="img-frame ${extraClass}"><div class="img-placeholder">image</div></div>`;
      }
      return `
        <div class="img-frame ${extraClass}" data-draggable-image="true">
          <img src="${image.src}" alt="uploaded image" style="${imageTransformStyle(image)}" draggable="false" />
        </div>
      `;
    }

    function renderThumbImage(page) {
      const image = getImage(page);
      if (!image || !image.src) return '';
      return `<img src="${image.src}" alt="thumb" style="${imageTransformStyle(image)}" draggable="false" />`;
    }

    function attachImageInteractions() {
      const page = appState.pages[appState.currentPage];
      const imageState = getImage(page);
      if (!imageState || !imageState.src) return;

      const frame = canvas.querySelector('[data-draggable-image="true"]');
      if (!frame) return;

      frame.onpointerdown = (e) => {
        activeDrag = {
          pointerId: e.pointerId,
          startX: e.clientX,
          startY: e.clientY,
          imageX: imageState.x,
          imageY: imageState.y
        };
        frame.classList.add('dragging');
        try { frame.setPointerCapture(e.pointerId); } catch (_) {}
      };

      frame.onpointermove = (e) => {
        if (!activeDrag || activeDrag.pointerId !== e.pointerId) return;
        const dx = e.clientX - activeDrag.startX;
        const dy = e.clientY - activeDrag.startY;
        imageState.x = Math.round(activeDrag.imageX + dx);
        imageState.y = Math.round(activeDrag.imageY + dy);
        renderPreviewOnly(false);
      };

      const endDrag = (e) => {
        if (!activeDrag) return;
        if (e && activeDrag.pointerId !== e.pointerId) return;
        activeDrag = null;
        frame.classList.remove('dragging');
        queueSave();
      };

      frame.onpointerup = endDrag;
      frame.onpointercancel = endDrag;
      frame.onlostpointercapture = () => {
        activeDrag = null;
        frame.classList.remove('dragging');
      };

      frame.onwheel = (e) => {
        e.preventDefault();
        const delta = -e.deltaY * 0.001;
        imageState.zoom = Math.min(2.5, Math.max(1, +(imageState.zoom + delta).toFixed(2)));
        renderPreviewOnly(false);
        queueSave();
      };
    }

    function renderPageList() {
      pageList.innerHTML = appState.pages.map((page, index) => `
        <div class="page-card ${index === appState.currentPage ? 'active' : ''}" data-index="${index}">
          <div class="page-top">
            <div>
              <div class="page-num">PAGE ${index + 1}</div>
              <div class="page-layout">${escapeHtml(getLayoutLabel(page.layout))}</div>
            </div>
            <div class="page-actions">
              <button class="mini move-up" data-index="${index}" type="button">↑</button>
              <button class="mini move-down" data-index="${index}" type="button">↓</button>
              <button class="mini delete-page" data-index="${index}" type="button">삭제</button>
            </div>
          </div>
          <div class="page-title">${escapeHtml(getPageDisplayTitle(page, index))}</div>
        </div>
      `).join('');

      pageList.querySelectorAll('.page-card').forEach(card => {
        card.addEventListener('click', (e) => {
          if (e.target.closest('.move-up') || e.target.closest('.move-down') || e.target.closest('.delete-page')) return;
          appState.currentPage = Number(card.dataset.index);
          render();
        });
      });

      pageList.querySelectorAll('.move-up').forEach(button => {
        button.addEventListener('click', (e) => {
          e.stopPropagation();
          const index = Number(button.dataset.index);
          if (index === 0) return;
          [appState.pages[index - 1], appState.pages[index]] = [appState.pages[index], appState.pages[index - 1]];
          appState.currentPage = index - 1;
          queueSave();
          render();
        });
      });

      pageList.querySelectorAll('.move-down').forEach(button => {
        button.addEventListener('click', (e) => {
          e.stopPropagation();
          const index = Number(button.dataset.index);
          if (index === appState.pages.length - 1) return;
          [appState.pages[index + 1], appState.pages[index]] = [appState.pages[index], appState.pages[index + 1]];
          appState.currentPage = index + 1;
          queueSave();
          render();
        });
      });

      pageList.querySelectorAll('.delete-page').forEach(button => {
        button.addEventListener('click', (e) => {
          e.stopPropagation();
          if (appState.pages.length === 1) return;
          const index = Number(button.dataset.index);
          appState.pages.splice(index, 1);
          if (appState.currentPage >= appState.pages.length) {
            appState.currentPage = appState.pages.length - 1;
          }
          queueSave();
          render();
        });
      });
    }

    function imageControlsHtml(page) {
      const image = getImage(page);
      if (!image) return '';
      return `
        <label>이미지 업로드
          <input type="file" id="field-image-file" accept="image/*" />
        </label>
        <label>또는 이미지 URL

        <div class="toolbar">

        </div>
      `;
    }

            <select id="field-theme">
              <option value="light" ${page.theme === 'light' ? 'selected' : ''}>Light gray</option>
              <option value="dark" ${page.theme === 'dark' ? 'selected' : ''}>Dark gray</option>
            </select>
          </label>
        `;
      }

      if (page.layout === 'definition-overlay') {
        html += `
          <label>한국어 제목
            <input type="text" id="field-koreanTitle" value="${escapeHtml(page.koreanTitle || '')}" />
          </label>
          <label>영어/발음 표기
            <input type="text" id="field-englishTitle" value="${escapeHtml(page.englishTitle || '')}" />
          </label>
          <label>설명
            <textarea id="field-description">${escapeHtml(page.description || '')}</textarea>
          </label>
          ${textScaleControlHtml(page)}
          ${imageControlsHtml(page)}
        `;
      }

      if (page.layout === 'split-left' || page.layout === 'split-right' || page.layout === 'top-title-image') {
        html += `
          <label>키워드 제목
            <input type="text" id="field-title" value="${escapeHtml(page.title || '')}" />
          </label>
          <label>설명
            <textarea id="field-details">${escapeHtml(page.details || '')}</textarea>
          </label>
          ${textScaleControlHtml(page)}
          ${imageControlsHtml(page)}
        `;
      }

      if (page.layout === 'ending') {
        html += `
          <label>마무리 카피
            <textarea id="field-copy">${escapeHtml(page.copy || '')}</textarea>
          </label>
          <label>핸들 / CTA
            <textarea id="field-handle">${escapeHtml(page.handle || '')}</textarea>
          </label>
          ${textScaleControlHtml(page)}
          <label>배경 톤
            <select id="field-theme">
              <option value="dark" ${page.theme === 'dark' ? 'selected' : ''}>Dark gray</option>
              <option value="light" ${page.theme === 'light' ? 'selected' : ''}>Light gray</option>

    function bindTextInput(id, key) {
      const element = document.getElementById(id);
      if (!element) return;
      const page = appState.pages[appState.currentPage];
      element.addEventListener('input', (e) => {
        page[key] = e.target.value;
        queueSave();
        renderPreviewOnly();
        renderPageList();
      });
    }

    function bindEditorEvents() {
      const page = appState.pages[appState.currentPage];

      const layoutElement = document.getElementById('field-layout');
      layoutElement.addEventListener('change', (e) => {
        const oldPage = appState.pages[appState.currentPage];
        const nextPage = createPage(e.target.value);

        const carryKeys = ['koreanTitle', 'englishTitle', 'description', 'title', 'details', 'copy', 'handle', 'theme', 'textScale'];
        carryKeys.forEach(key => {
          if (key in nextPage && oldPage[key] !== undefined && oldPage[key] !== null && oldPage[key] !== '') {
            nextPage[key] = oldPage[key];
          }
        });

        if ('image' in nextPage && 'image' in oldPage && oldPage.image) {
          nextPage.image = structuredClone(oldPage.image);
        }

        appState.pages[appState.currentPage] = nextPage;
        queueSave();
        render();
      });

      bindTextInput('field-koreanTitle', 'koreanTitle');
      bindTextInput('field-englishTitle', 'englishTitle');
      bindTextInput('field-description', 'description');
      bindTextInput('field-title', 'title');
      bindTextInput('field-details', 'details');
      bindTextInput('field-copy', 'copy');
      bindTextInput('field-handle', 'handle');

      const textScaleInput = document.getElementById('field-text-scale');
      const textScaleValue = document.getElementById('textScaleValue');
      if (textScaleInput) {
        textScaleInput.addEventListener('input', (e) => {
          page.textScale = Number(e.target.value);
          if (textScaleValue) textScaleValue.textContent = page.textScale.toFixed(2);
          queueSave();
          renderPreviewOnly();
          renderPageList();
        });
      }

      const themeElement = document.getElementById('field-theme');
      if (themeElement) {
        themeElement.addEventListener('change', (e) => {
          page.theme = e.target.value;
          queueSave();
          renderPreviewOnly();
        });
      }

      const image = getImage(page);
      if (image) {
        const fileInput = document.getElementById('field-image-file');
        const urlInput = document.getElementById('field-image-url');
        const removeImageBtn = document.getElementById('removeImageBtn');
        const resetImageAdjustBtn = document.getElementById('resetImageAdjustBtn');

        fileInput?.addEventListener('change', (e) => {
          const file = e.target.files?.[0];
          if (!file) return;
          const reader = new FileReader();
          reader.onload = () => {
            image.src = reader.result;
            queueSave();
            renderPreviewOnly();
            if (urlInput) urlInput.value = image.src;
          };
          reader.readAsDataURL(file);
        });

        urlInput?.addEventListener('input', (e) => {
          image.src = e.target.value;
          queueSave();
          renderPreviewOnly();
        });

        removeImageBtn?.addEventListener('click', () => {
          image.src = '';
          image.zoom = 1;
          image.x = 0;
          image.y = 0;
          queueSave();
          render();
        });

        resetImageAdjustBtn?.addEventListener('click', () => {
          image.zoom = 1;
          image.x = 0;
          image.y = 0;
          queueSave();
          renderPreviewOnly();
        });
      }
    }

    function renderSlide(page) {
      const scale = page.textScale || 1;

      if (page.layout === 'korean-hero') {
        return `
          <div class="slide ${page.theme || 'light'}">
            <div class="hero">
              <div class="hero-inner" style="transform: scale(${scale});">
                <div class="hero-title">${multilineToHtml(page.koreanTitle || '추구미')}</div>
              </div>
            </div>
          </div>
        `;
      }

      if (page.layout === 'definition-overlay') {
        const image = getImage(page);
        const bg = image && image.src
          ? `<div class="definition-bg">${renderImageBox(image)}</div>`
          : `<div class="definition-bg" style="background:#7b7b7b"></div>`;
        return `
          <div class="slide dark">
            ${bg}
            <div class="definition-overlay"></div>
            <div class="definition-content">
              <div class="definition-inner" style="transform: scale(${scale});">
                <div class="def-ko">${multilineToHtml(page.koreanTitle || '추구미')}</div>
                <div class="def-en">${multilineToHtml(page.englishTitle || '[chugumi]')}</div>
                <div class="bar"></div>
                <div class="def-desc">${multilineToHtml(page.description || '')}</div>
              </div>
            </div>
          </div>
        `;
      }

      if (page.layout === 'split-left' || page.layout === 'split-right') {
        const imageHtml = renderImageBox(getImage(page));
        const textHtml = `
          <div class="split-text">
            <div class="split-text-inner" style="transform: scale(${scale});">
              <div class="hash-title">${multilineToHtml(page.title || '#Keyword')}</div>
              <div class="details">${multilineToHtml(page.details || '')}</div>
            </div>
          </div>
        `;
        return `
          <div class="slide light">
            <div class="split">
              ${page.layout === 'split-left' ? imageHtml + textHtml : textHtml + imageHtml}
            </div>
          </div>
        `;
      }

      if (page.layout === 'top-title-image') {
        return `
          <div class="slide light">
            <div class="top-layout">
              <div class="top-text">
                <div class="top-inner" style="transform: scale(${scale});">
                  <div class="top-title">${multilineToHtml(page.title || '#Keyword')}</div>
                  <div class="top-desc">${multilineToHtml(page.details || '')}</div>
                </div>
              </div>
              <div class="top-image">${renderImageBox(getImage(page))}</div>
            </div>
          </div>
        `;
      }

      if (page.layout === 'ending') {
        return `
          <div class="slide ${page.theme || 'dark'}">
            <div class="ending">
              <div class="ending-inner" style="transform: scale(${scale});">
                <div class="ending-copy">${multilineToHtml(page.copy || '')}</div>
                <div class="bar" style="height:80px"></div>
                <div class="ending-handle">${multilineToHtml(page.handle || '')}</div>
              </div>
            </div>
          </div>
        `;
      }

      return '<div class="slide light"></div>';
    }

    function renderThumbs() {
      thumbs.innerHTML = appState.pages.map((page, index) => `
        <button class="thumb ${index === appState.currentPage ? 'active' : ''}" data-index="${index}">
          <div class="thumb-mini">${renderThumbImage(page)}</div>
          <span class="thumb-num">${index + 1}</span>
        </button>
      `).join('');

      thumbs.querySelectorAll('.thumb').forEach(button => {
        button.addEventListener('click', () => {
          appState.currentPage = Number(button.dataset.index);
          render();
        });
      });
    }

    function renderPreviewOnly(reattach = true) {
      const page = appState.pages[appState.currentPage];
      previewTitle.textContent = `PAGE ${appState.currentPage + 1} · ${getLayoutLabel(page.layout)}`;
      canvas.innerHTML = renderSlide(page);
      renderThumbs();
      if (reattach) attachImageInteractions();
    }

    function render() {
      renderPageList();
      renderEditor();
      renderPreviewOnly();
    }

    addPageBtn.addEventListener('click', () => {
      appState.pages.push(createPage('korean-hero'));
      appState.currentPage = appState.pages.length - 1;
      queueSave();
      render();
    });

    duplicatePageBtn.addEventListener('click', () => {
      const clone = structuredClone(appState.pages[appState.currentPage]);
      appState.pages.splice(appState.currentPage + 1, 0, clone);
      appState.currentPage += 1;
      queueSave();
      render();
    });

    resetAllBtn.addEventListener('click', () => {
      const ok = window.confirm('정말 전체 작업을 초기화할까요? 저장된 내용도 함께 지워집니다.');
      if (!ok) return;
      localStorage.removeItem(STORAGE_KEY);
      const fresh = defaultState();
      appState.currentPage = fresh.currentPage;
      appState.pages = fresh.pages;
      setSaveStatus('초기화 완료');
      render();
    });

    render();
    setSaveStatus(localStorage.getItem(STORAGE_KEY) ? '자동저장 불러옴' : '자동저장 대기');
  </script>
